							}

					}
				}
			}

			##############

				limit = {

						array = global.gw_target_supporters_array
						value = THIS
					}

							is_in_array = {
								array = global.gw_target_array

							}
						}
						add_to_faction = PREV
					}
				}

					random_country = {
						limit = {
							is_in_array = {
								array = global.gw_target_array
								value = THIS
							}
						}
						add_to_faction = PREV
					}
				}
			}

			###########
			### WAR ###
			###########

			
			random_country = {
				limit = {
					is_in_array = {
						array = global.gw_target_array
						value = THIS
					}
				}
				random_country = {
					limit = {
						is_in_array = {
							array = global.gw_initiator_array
							value = THIS
						}
					}
					declare_war_on = {
						target = PREV
						type = topple_government
					}
				}
			}

			random_country = {
				limit = {
					is_in_array = {
						array = global.gw_target_array
						value = THIS
					}
				}
				every_country = {
					limit = {
						is_in_array = {
							array = global.gw_initiator_supporters_array
							value = THIS
						}		
					}
					declare_war_on = {
						target = PREV
						type = topple_government
					}
				}
			}

		}
